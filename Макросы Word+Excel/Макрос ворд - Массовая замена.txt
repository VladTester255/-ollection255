Sub МассоваяЗамена()
'
' Макрос2 Макрос
'
'
    Selection.Find.Replacement.ClearFormatting
    With Selection.Find
        .Text = "([ ^s]){2;}"
        .Replacement.Text = "\1"
        .Forward = True
        .Wrap = wdFindContinue
        .Format = False
        .MatchCase = False
        .MatchWholeWord = False
        .MatchWildcards = True
        .MatchSoundsLike = False
        .MatchAllWordForms = False
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
    With Selection.Find
        .Text = "^s "
        .Replacement.Text = "^s"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
    With Selection.Find
        .Text = " ^s"
        .Replacement.Text = "^s"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
    With Selection.Find
        .Text = "ГОСТ ([0-9])"
        .Replacement.Text = "ГОСТ^s\1"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
    With Selection.Find
        .Text = "ГОСТ Р ([0-9])"
        .Replacement.Text = "ГОСТ^sР^s\1"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "ГОСТ Р МЭК ([0-9])"
        .Replacement.Text = "ГОСТ^sР^sМЭК^s\1"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "(п{1;2}). ([0-9])"
        .Replacement.Text = "\1^s\2"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "(р{1;2}.) ([0-9])"
        .Replacement.Text = "\1^s\2"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "( [улдгп]{1;2}.) ([0-9А-Я])"
        .Replacement.Text = "\1^s\2"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "([а-я]) (обл.)"
        .Replacement.Text = "\1^s\2"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "°С"
        .Replacement.Text = "°C"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "от[ ^s]([плюсминус]{4;5})[ ^s]([0-9\.,]@)[ ^s]([°C]{1;3}) "
        .Replacement.Text = "от^s\1^s\2^s\3 "
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = " до[ ^s]([плюсминус]{4;5})[ ^s]([0-9\.,]@)[ ^s]([°C]{2})"
        .Replacement.Text = " до^s\1^s\2^s\3"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "от[ ^s]([0-9\.,]@)[ ^s]([кПа%]{1;3}) "
        .Replacement.Text = "от^sплюс^s\1^s\2 "
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = " до[ ^s]([0-9\.,]@)[ ^s]([кПа%]{1;3})"
        .Replacement.Text = " до^sплюс^s\1^s\2"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "[ ^s][-–—][ ^s]"
        .Replacement.Text = "^s–^s"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "зав.[ ^s]"
        .Replacement.Text = "зав.^s"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "№[ ^s]*"
        .Replacement.Text = "№^s"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
    
    Selection.Find.Replacement.ClearFormatting
        With Selection.Find
        .Text = "([0-9])[ ^s](с[!а-я])"
        .Replacement.Text = "\1^s\2"
    End With
    Selection.Find.Execute Replace:=wdReplaceAll
End Sub
